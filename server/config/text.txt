

-- Tạo bảng Category
CREATE TABLE Category (
    id INT PRIMARY KEY,
    name VARCHAR(255) NOT NULL
);

-- Tạo bảng Product
CREATE TABLE Product (
    id INT PRIMARY KEY,
    category_id INT,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    product_image VARCHAR(255),
    FOREIGN KEY (category_id) REFERENCES Category(id)
);

-- Tạo bảng Variation
CREATE TABLE Variation (
    id INT PRIMARY KEY,
    product_id INT,
    SKU VARCHAR(50) NOT NULL,
    qty_in_stock INT,
    product_image VARCHAR(255),
    price DECIMAL(10, 2),
    FOREIGN KEY (product_id) REFERENCES Product(id)
);

-- Tạo bảng Option_p
CREATE TABLE Option_p (
    id INT PRIMARY KEY,
    category_id INT,
    name VARCHAR(255) NOT NULL,
    FOREIGN KEY (category_id) REFERENCES Category(id)
);

-- Tạo bảng Option_value
CREATE TABLE Option_value (
    id INT PRIMARY KEY,
    option_id INT,
    value VARCHAR(255) NOT NULL,
    FOREIGN KEY (option_id) REFERENCES Option_p(id)
);

-- Tạo bảng Combination
CREATE TABLE Combination (
    id INT PRIMARY KEY,
    variation_id INT,
    option_value_id INT,
    FOREIGN KEY (variation_id) REFERENCES Variation(id),
    FOREIGN KEY (option_value_id) REFERENCES Option_value(id)
);
